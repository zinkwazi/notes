==== SSH ====
_IP=$(/bin/hostname -I) || true
TPORT=1088

sudo -H -u pi /usr/bin/wget -qO- "http://www.foo.com/pihome/index.php?tport=$TPORT&ip=$_IP" &> /dev/null
ssh -R 8080:192.168.1.2:3389 me@office.com -p 8008
